    <script>
        // Quadratic Equation Solver Functions
        function solveQuadratic() {
            // Get input values
            const a = parseFloat(document.getElementById('a').value);
            const b = parseFloat(document.getElementById('b').value);
            const c = parseFloat(document.getElementById('c').value);
            
            // Reset errors
            document.getElementById('errorA').style.display = 'none';
            document.getElementById('errorB').style.display = 'none';
            document.getElementById('errorC').style.display = 'none';
            
            // Validate inputs
            let isValid = true;
            
            if (isNaN(a)) {
                document.getElementById('errorA').textContent = 'Please enter a valid number for coefficient a';
                document.getElementById('errorA').style.display = 'block';
                isValid = false;
            } else if (a === 0) {
                document.getElementById('errorA').textContent = 'Coefficient a cannot be zero';
                document.getElementById('errorA').style.display = 'block';
                isValid = false;
            }
            
            if (isNaN(b)) {
                document.getElementById('errorB').textContent = 'Please enter a valid number for coefficient b';
                document.getElementById('errorB').style.display = 'block';
                isValid = false;
            }
            
            if (isNaN(c)) {
                document.getElementById('errorC').textContent = 'Please enter a valid number for coefficient c';
                document.getElementById('errorC').style.display = 'block';
                isValid = false;
            }
            
            if (!isValid) return;
            
            // Calculate discriminant
            const discriminant = b * b - 4 * a * c;
            
            // Display results
            const resultDiv = document.getElementById('quadraticResult');
            resultDiv.style.display = 'block';
            
            let resultHTML = `
                <h3>Equation: ${formatCoefficient(a)}x² ${formatCoefficient(b, true)}x ${formatCoefficient(c, true)} = 0</h3>
                <div class="equation">Discriminant (D) = b² - 4ac = ${discriminant.toFixed(4)}</div>
            `;
            
            if (discriminant > 0) {
                // Two distinct real roots
                const root1 = (-b + Math.sqrt(discriminant)) / (2 * a);
                const root2 = (-b - Math.sqrt(discriminant)) / (2 * a);
                
                resultHTML += `
                    <p class="info">Nature of roots: Two distinct real roots (D > 0)</p>
                    <p class="success">Root 1 (x₁) = ${root1.toFixed(4)}</p>
                    <p class="success">Root 2 (x₂) = ${root2.toFixed(4)}</p>
                `;
            } else if (discriminant === 0) {
                // One real repeated root
                const root = -b / (2 * a);
                
                resultHTML += `
                    <p class="info">Nature of roots: One real repeated root (D = 0)</p>
                    <p class="success">Root (x) = ${root.toFixed(4)}</p>
                `;
            } else {
                // Two complex conjugate roots
                const realPart = -b / (2 * a);
                const imaginaryPart = Math.sqrt(-discriminant) / (2 * a);
                
                resultHTML += `
                    <p class="info">Nature of roots: Two complex conjugate roots (D < 0)</p>
                    <p class="success">Root 1 (x₁) = ${realPart.toFixed(4)} + ${imaginaryPart.toFixed(4)}i</p>
                    <p class="success">Root 2 (x₂) = ${realPart.toFixed(4)} - ${imaginaryPart.toFixed(4)}i</p>
                `;
            }
            
            resultDiv.innerHTML = resultHTML;
        }
        
        function clearQuadratic() {
            document.getElementById('a').value = '';
            document.getElementById('b').value = '';
            document.getElementById('c').value = '';
            document.getElementById('quadraticResult').style.display = 'none';
            
            // Clear errors
            document.getElementById('errorA').style.display = 'none';
            document.getElementById('errorB').style.display = 'none';
            document.getElementById('errorC').style.display = 'none';
        }
        
        // Helper function to format coefficients in the equation display
        function formatCoefficient(value, withSign = false) {
            if (value === 0) return withSign ? '+ 0' : '0';
            
            let sign = value < 0 ? '-' : (withSign ? '+' : '');
            let absValue = Math.abs(value);
            
            return ${sign} ${absValue === 1 ? '' : absValue};
        }
        
        // Grade Converter Functions
        function convertGrade() {
            // Get input value
            const score = parseInt(document.getElementById('score').value);
            
            // Reset error
            document.getElementById('errorScore').style.display = 'none';
            
            // Validate input
            if (isNaN(score)) {
                document.getElementById('errorScore').textContent = 'Please enter a valid score';
                document.getElementById('errorScore').style.display = 'block';
                return;
            }
            
            if (score < 0 || score > 100) {
                document.getElementById('errorScore').textContent = 'Score must be between 0 and 100';
                document.getElementById('errorScore').style.display = 'block';
                return;
            }
            
            // Determine grade
            let grade;
            if (score >= 85) grade = 'A+';
            else if (score >= 75) grade = 'A';
            else if (score >= 65) grade = 'B+';
            else if (score >= 60) grade = 'B';
            else if (score >= 55) grade = 'C+';
            else if (score >= 50) grade = 'C';
            else grade = 'D';
            
            // Display result
            const resultDiv = document.getElementById('gradeResult');
            resultDiv.style.display = 'block';
            resultDiv.innerHTML = `
                <h3>Grade Conversion Result</h3>
                <p class="success">Score ${score} → Grade ${grade}</p>
                <p class="info">Based on the provided grading scale</p>
            `;
        }
        
        function clearGrade() {
            document.getElementById('score').value = '';
            document.getElementById('gradeResult').style.display = 'none';
            document.getElementById('errorScore').style.display = 'none';
        }
